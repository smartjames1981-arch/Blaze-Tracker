    <div style="text-align: left; padding-bottom: 10px; font-weight: bold;">BENCH</div>

<script>
    let seconds = 0;
    let timerInterval;
    let isRunning = false;
    let score = 0;

    // Based on your team needs:
    const players = [
        { name: "Quinn", time: 0, goals: 0, onField: false },
        { name: "Cole", time: 0, goals: 0, onField: false },
        { name: "MJ", time: 0, goals: 0, onField: false },
        { name: "Laiken", time: 0, goals: 0, onField: false },
        { name: "Micah", time: 0, goals: 0, onField: false },
        { name: "Caleb", time: 0, goals: 0, onField: false },
        { name: "Ethan", time: 0, goals: 0, onField: false }
    ];

    let fieldSlots = { st: null, lw: null, mf: null, rw: null, cb1: null, cb2: null, gk: null };

    function renderBench() {
        const list = document.getElementById('benchList');
        list.innerHTML = '';
        players.forEach((p, index) => {
            if (!p.onField) {
                const card = document.createElement('div');
                card.className = 'player-card';
                card.innerHTML = `<span>${p.name} ${"âš½".repeat(p.goals)}</span><span class="playing-time">${formatTime(p.time)}</span>`;
                card.onclick = () => subIn(index);
                list.appendChild(card);
            }
        });
    }

    function formatTime(s) {
        const mins = Math.floor(s / 60);
        const secs = s % 60;
        return `${mins}m ${secs}s`;
    }

    function toggleTimer() {
        isRunning = !isRunning;
        document.getElementById('startBtn').innerText = isRunning ? "PAUSE" : "START";
        if (isRunning) {
            timerInterval = setInterval(() => {
                seconds++;
                document.getElementById('timer').innerText = new Date(seconds * 1000).toISOString().substr(14, 5);
                players.filter(p => p.onField).forEach(p => p.time++);
                renderBench();
            }, 1000);
        } else {
            clearInterval(timerInterval);
        }
    }

    function subIn(playerIndex) {
        const slot = prompt("Enter position (st, lw, mf, rw, cb1, cb2, gk):").toLowerCase();
        if (fieldSlots[slot] === null) {
            players[playerIndex].onField = true;
            fieldSlots[slot] = players[playerIndex];
            document.getElementById(`name-${slot}`).innerText = players[playerIndex].name;
            renderBench();
        }
    }

    function subOut(slot) {
        if (fieldSlots[slot]) {
            if (confirm(`Goal for ${fieldSlots[slot].name}?`)) {
                fieldSlots[slot].goals++;
                score++;
                document.getElementById('score').innerText = score;
            } else {
                fieldSlots[slot].onField = false;
                fieldSlots[slot] = null;
                document.getElementById(`name-${slot}`).innerText = "EMPTY";
            }
            renderBench();
        }
    }

    renderBench();
</script>
